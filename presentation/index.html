Regular expressions
Create a regular expression object 

          regexp = new RegExp("pattern", "flags");
          

            regexp = /pattern/; // no flags
            regexp = /pattern/gmi; // with flags g,m and i
Flags: i, g, m, s, u, y, d
str.match(regexp) 

              let str = "There is the house where my family lives";

              alert( str.match(/ere/gi) ); 
              // ere, ere (an array of 2 substrings that match)
str.replace(regexp, replacement) 

              // no flag g
              alert( "To be or not to be".replace(/to/i, "I") ); 
              // I be or not to be
              
              // with flag g
              alert( "To be or not to be".replace(/to/ig, "I") ); 
              // I be or not I be
regexp.test(str) 

              let str = "To be or not to be";
              let regexp = /BE/i;

              alert( regexp.test(str) ); // true
Character classes: \d, \D, \s, \S, \w, \W, . 

              let str = "+1(234)-567-89-12";

              alert( str.match(/\d/) ); // 1

              alert( str.match(/\d/g) ); 
              // array of matches: [1,2,3,4,5,6,7,8,9,1,2]
Anchors: ^ and $ 

            let str = "There is the house where my family lives";
            
            alert( /^There/.test(str) ); // true

            alert( /lives$/.test(str) ); // true
Escaping: \

Special characters: [ ] { } ( ) \ ^ $ . | ? * +


            alert( "Chapter 7.2".match(/\d\.\d/) ); // 7.2
Sets and ranges [...]
Sets 

              alert( "Ava avi".match(/av[ai]/gi) ); // "Ava", "avi"
Ranges 

              alert( "My family".match(/i[0-9a-z][0-9a-z]/g) ); // ily
Excluding ranges 

              alert( "example5@gmail.com".match(/[^\d\sA-Z]/gi) ); // @ and .
Quantifiers +, *, ? and {n}
{n} 

              alert( "My number 12345".match(/\d{5}/) ); //  "12345"

              alert( "My number 1234".match(/\d{3,5}/) ); //  "1234"
              
              alert( "My number 1234567".match(/\d{3,}/) ); //  "1234567"
+ 

              alert( "+1(234)-567-89-12".match(/\d+/g) ); //  1,234,567,89,12
? 

              alert( "Should I write color or colour?".match(/colou?r/g) ); 
              // color, colour
* 

              alert( "100 10 1".match(/\d0*/g) ); // 100, 10, 1
Capturing groups (...) 

            alert( 'Gogogo now!'.match(/(go)+/ig) ); // "Gogogo"
Alternation | 

            let regexp = /html|php|css|java(script)?/gi;

            let str = "First HTML appeared, then CSS, then JavaScript";

            alert( str.match(regexp) ); // 'HTML', 'CSS', 'JavaScript'
